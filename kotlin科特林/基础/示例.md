```kotlin
open class Shape{
    fun m(){
        println("hello world")
    }
}

// 写类跟写函数一样。 每个类中都有默认构造函数
class Main(var height: Double, var length: Double): Shape() // 继承
{
    var perimeter = (height + length) * 2
}

fun main() {
    var a = Main(100.0,200.0)
    println(a.perimeter)
    println(a.m())
}
```
